  imports: [CommonModule],
  providers: [ScreenPointsService],
})
export class ScreenPointsModule {
  static forRoot(): ModuleWithProviders<any> {

      ngModule: ScreenPointsService,
      providers: [ScreenPointsService],
    };
  }
